2. **Replace JS state** (around line 192): Remove `let activeFamily = 'all'`. Add exclusion Set and localStorage functions:

```javascript
var _mapExclusions = new Set();
var MAP_EXCLUSIONS_KEY = 'custard:map:exclusions';

function restoreMapExclusions() {
  try {
    var raw = localStorage.getItem(MAP_EXCLUSIONS_KEY);
    if (raw) {
      var arr = JSON.parse(raw);
      if (Array.isArray(arr)) _mapExclusions = new Set(arr);
    }
  } catch (e) {}
}

function saveMapExclusions() {
  try {
    var arr = [];
    _mapExclusions.forEach(function(key) { arr.push(key); });
    localStorage.setItem(MAP_EXCLUSIONS_KEY, JSON.stringify(arr));
  } catch (e) {}
}
```

```javascript
function isStoreExcluded(store) {
  if (_mapExclusions.size === 0) return false;
  if (!store.flavor) return false;
  var normalized = CustardPlanner.normalize(store.flavor);
  var families = flavorToFamilies[normalized];
  if (!families) return false;
  for (var i = 0; i < families.length; i++) {
    if (_mapExclusions.has(families[i])) return true;
  }
  return false;
}
```

7. **Rewrite applyFamilyFilter** to use exclusion dimming (opacity 0.15 for excluded, 1 for included) -- per research recommendation to dim rather than hide:

```javascript
function applyFamilyFilter() {
  if (!window._allMarkers) return;
  for (var i = 0; i < window._allMarkers.length; i++) {
    var entry = window._allMarkers[i];
    var excluded = isStoreExcluded(entry.store);
    entry.marker.setOpacity(excluded ? 0.15 : 1);
  }
}
```
