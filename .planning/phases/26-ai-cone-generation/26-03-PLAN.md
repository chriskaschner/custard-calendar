**Step 1: Read the approved manifest.**

Read `docs/assets/ai-generation-manifest.json`. For each flavor with `status: "approved"`, identify the `selected` candidate filename.

**Step 2: Copy approved candidates to final location.**

For each approved flavor, copy the post-processed selected candidate from `docs/assets/ai-candidates/{slug}/processed/{selected}` to `docs/assets/cones/{slug}.png`, overwriting the existing SVG-rasterized PNG.

This can be done with a small Node.js script or shell loop:
```bash
node -e "
const manifest = require('./docs/assets/ai-generation-manifest.json');
const fs = require('fs');
const path = require('path');
for (const [key, entry] of Object.entries(manifest.flavors)) {
  if (entry.status !== 'approved') { console.error('NOT APPROVED:', key); continue; }
  const src = path.join('docs/assets/ai-candidates', entry.slug, 'processed', entry.selected);
  const dst = path.join('docs/assets/cones', entry.slug + '.png');
  fs.copyFileSync(src, dst);
  console.log('copied', entry.slug);
}
"
```
