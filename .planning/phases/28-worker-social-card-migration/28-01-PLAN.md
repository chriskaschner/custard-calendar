<objective>
Migrate social-card.js from rendering inline SVG cones (via renderConeHDSVG) to embedding L5 AI PNG cone images as base64 data URIs within the SVG card output.

Purpose: OG social cards shared on Twitter/Facebook/LinkedIn display the high-quality AI-generated cone art instead of the low-resolution algorithmic SVG pixel cones. This is the integration step (INT-04) that connects the Worker's social card pipeline to the L5 PNG assets deployed in Phase 26.

Output: Updated social-card.js that fetches cone PNGs from GitHub Pages, converts to base64, and embeds as SVG `<image>` elements. Updated tests validating PNG embedding.
</objective>

    ```javascript
    const CONE_PNG_BASE = 'https://custard.chriskaschner.com/assets/cones';

    async function fetchConePngBase64(flavorName) {
      const slug = flavorToSlug(flavorName);
      try {
        const resp = await fetch(`${CONE_PNG_BASE}/${slug}.png`);
        if (!resp.ok) return null;
        const buffer = await resp.arrayBuffer();
        const bytes = new Uint8Array(buffer);
        let binary = '';
        for (let i = 0; i < bytes.length; i++) binary += String.fromCharCode(bytes[i]);
        return btoa(binary);
      } catch {
        return null;
      }
    }
    ```

    ```javascript
    async function renderConeEmbed(flavorName, x, y, width, height) {
      const b64 = await fetchConePngBase64(flavorName);
      if (b64) {
        return `<image x="${x}" y="${y}" width="${width}" height="${height}" href="data:image/png;base64,${b64}" preserveAspectRatio="xMidYMid meet"/>`;

    }
    ```
