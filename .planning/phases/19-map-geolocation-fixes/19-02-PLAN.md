**In map.html** -- Add nearest-store detection and highlighting:

1. **Add nearest store detection function** (after the `updatePositionDot` function created in Plan 01):
```javascript
let nearestStoreSlug = null; // tracks current nearest store for dynamic updates

function detectAndHighlightNearest() {
  if (userLat === null || userLon === null || !window._allMarkers || window._allMarkers.length === 0) {
    nearestStoreSlug = null;
    return;
  }
  let minDist = Infinity;
  let nearestEntry = null;
  for (let i = 0; i < window._allMarkers.length; i++) {
    const entry = window._allMarkers[i];
    const dist = haversine(userLat, userLon, entry.store._lat, entry.store._lon);
    if (dist < minDist) {
      minDist = dist;
      nearestEntry = entry;
    }
  }
  nearestStoreSlug = nearestEntry ? (nearestEntry.store.slug || null) : null;

  // Apply CSS class to nearest marker, remove from all others
  for (let i = 0; i < window._allMarkers.length; i++) {
    const entry = window._allMarkers[i];
    const el = entry.marker.getElement && entry.marker.getElement();
    if (!el) continue;
    const markerDiv = el.querySelector('.flavor-map-marker');
    if (!markerDiv) continue;
    if (entry === nearestEntry) {
      markerDiv.classList.add('flavor-map-marker-nearest');
    } else {
      markerDiv.classList.remove('flavor-map-marker-nearest');
    }
  }
}
```
