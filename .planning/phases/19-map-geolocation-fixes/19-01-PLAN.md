```javascript
async function reverseGeocode(lat, lng) {
  try {
    const resp = await fetch(`https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lng}&format=json&zoom=10`);
    const data = await resp.json();
    const addr = data.address || {};
    const city = addr.city || addr.town || addr.village || addr.county || '';
    const state = addr.state || '';
    return city && state ? `${city}, ${state}` : city || `${lat.toFixed(4)},${lng.toFixed(4)}`;
  } catch { return `${lat.toFixed(4)},${lng.toFixed(4)}`; }
}
```

```javascript
async function initLocation() {
  // Try GPS first (auto-request on page load per user decision)
  if (navigator.geolocation) {
    try {
      await new Promise((resolve, reject) => {
        navigator.geolocation.getCurrentPosition(
          async (pos) => {
            const lat = pos.coords.latitude;
            const lon = pos.coords.longitude;
            // Center map on GPS position
            skipMoveHandler = true;
            map.setView([lat, lon], 11);
            // Show position dot
            updatePositionDot(lat, lon);
            // Reverse geocode to populate location input
            const placeName = await reverseGeocode(lat, lon);
            locationInput.value = placeName;
            // Start live tracking
            startWatchPosition();
            // Trigger search with the geocoded location
            doSearch();
            resolve();
          },
          (err) => {
            reject(err);
          },
          { enableHighAccuracy: true, timeout: 8000, maximumAge: 60000 }
        );
      });
      return; // GPS succeeded, done
```

```javascript
function startWatchPosition() {
  if (watchId !== null || !navigator.geolocation) return;
  watchId = navigator.geolocation.watchPosition(
    (pos) => {
      // Update dot position but do NOT re-center map (per user decision)
      updatePositionDot(pos.coords.latitude, pos.coords.longitude);
    },
    () => { /* watch error is non-fatal */ },
    { enableHighAccuracy: false, maximumAge: 30000 }
  );
}
```
