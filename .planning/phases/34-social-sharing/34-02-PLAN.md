    ```javascript
    function renderShareResult(archetype, flavorName) {
      // Hide quiz form sections
      var heroSection = document.querySelector('.quiz-hero');
      var panelSection = document.querySelector('.quiz-panel');
      if (heroSection) heroSection.hidden = true;
      if (panelSection) panelSection.hidden = true;

      // Populate result section
      if (els.resultTitle) els.resultTitle.textContent = archetype.name;
      if (els.resultFlavor) els.resultFlavor.textContent = flavorName;
      if (els.resultBlurb) els.resultBlurb.textContent = archetype.blurb || '';

      // Render cone art
      if (els.resultCone && typeof renderHeroCone === 'function') {
        renderHeroCone(flavorName, els.resultCone, 6);
      }

      // Mount share button with flavor-themed text
      if (els.resultShare) {
        els.resultShare.innerHTML = '';
        var shareBtn = document.createElement('button');
        shareBtn.className = 'share-btn result-share-btn';
        shareBtn.textContent = 'Share your result';
        var shareText = "I'm a " + archetype.name + ' (' + flavorName + ') -- what\'s your custard personality?';
        var shareUrl = window.location.href;
        shareBtn.addEventListener('click', function () {
          if (navigator.share) {
            navigator.share({ title: shareText, url: shareUrl }).catch(function () {});
          } else {
            navigator.clipboard.writeText(shareText + ' ' + shareUrl).then(function () {
              shareBtn.textContent = 'Link copied!';
              setTimeout(function () { shareBtn.textContent = 'Share your result'; }, 2000);
            }).catch(function () {
              window.prompt('Copy this link:', shareUrl);
            });
          }
        });
        els.resultShare.appendChild(shareBtn);
      }

      // Add "Take the quiz yourself" CTA button
      if (els.resultCtas) {
        var ctaBtn = document.createElement('a');
        ctaBtn.href = 'quiz.html';
        ctaBtn.className = 'btn btn-primary share-result-cta';
        ctaBtn.textContent = 'Take the quiz yourself';
        ctaBtn.addEventListener('click', function (e) {
          e.preventDefault();
          // Clear URL params and reload to fresh quiz state
          window.location.href = 'quiz.html';
        });
        els.resultCtas.appendChild(ctaBtn);
      }

      // Show result section
      els.resultSection.hidden = false;
    }
    ```

    Note: Do NOT use f-strings or template literals -- engine.js uses string concatenation throughout (ES5 compatibility pattern). All functions must use `function` declarations, not arrow functions.

    In the today-page.js IIFE, add a new function `handleFlavorParam()` that runs after the week strip has been rendered. Place it after the `renderWeekStrip` function definition (around line 450):

    ```javascript
    function handleFlavorParam() {
      var params = new URLSearchParams(window.location.search);
      var flavorParam = params.get('flavor');
      if (!flavorParam || !weekStrip) return;

      // Open the week-section details element if closed
      if (weekSection && weekSection.tagName === 'DETAILS') {
        weekSection.open = true;
      }

      // Find matching card in week strip
      var cards = weekStrip.querySelectorAll('.week-day-card');
      var targetCard = null;
      for (var i = 0; i < cards.length; i++) {
        var flavorEl = cards[i].querySelector('.week-day-flavor');
        if (flavorEl && flavorEl.textContent.trim().toLowerCase() === flavorParam.trim().toLowerCase()) {
          targetCard = cards[i];
          break;
        }
      }

      if (!targetCard) return;

      // Scroll to the card
      setTimeout(function () {
        targetCard.scrollIntoView({ behavior: 'smooth', block: 'center' });
      }, 300);

      // Apply highlight
      targetCard.classList.add('flavor-highlight');
      // Remove highlight after 4 seconds (3s visible + 1s fade)
      setTimeout(function () {
        targetCard.classList.remove('flavor-highlight');
      }, 4000);
    }
    ```
